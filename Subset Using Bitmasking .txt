#inlcude<iostream>
#include<cmath>
using namespace std;
int main()
{  int n;
   cin>>n;                                    // 1 2 3 4 5 6 ....
    for(int mask=0;mask<=pow(2,n)-1;mask++){  // A B C D E F ....
        int temp=mask;
        int i=0;
        while(temp>0){
            if((temp&1)==1)
            { cout<<(char)('A'+i)<<" "; }
            temp=temp>>1;
            i++;
        }
        cout<<endl;
    }
}
